507def _indentation(lexer, match, ctx): 

508 indentation = match.group(0) 

509 yield match.start(), Text, indentation 

510 ctx.last_indentation = indentation 

511 ctx.pos = match.end() 

512 

513 if hasattr(ctx, 'block_state') and ctx.block_state and \ 

514 indentation.startswith(ctx.block_indentation) and \ 

515 indentation != ctx.block_indentation: 

516 ctx.stack.append(ctx.block_state) 

517 else: 

518 ctx.block_state = None 

519 ctx.block_indentation = None 

520 ctx.stack.append('content') 

521 

522 

523def _starts_block(token, state): 

524 def callback(lexer, match, ctx): 

525 yield match.start(), token, match.group(0) 

526 

527 if hasattr(ctx, 'last_indentation'): 

528 ctx.block_indentation = ctx.last_indentation 

529 else: 

530 ctx.block_indentation = '' 

531 

532 ctx.block_state = state 

533 ctx.pos = match.end() 

534 

535 return callback
