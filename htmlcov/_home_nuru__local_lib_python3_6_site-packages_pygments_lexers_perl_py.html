484 def _build_word_match(words, boundary_regex_fragment=None, prefix='', suffix=''): 

485 if boundary_regex_fragment is None: 

486 return r'\b(' + prefix + r'|'.join(re.escape(x) for x in words) + \ 

487 suffix + r')\b' 

488 else: 

489 return r'(?<!' + boundary_regex_fragment + r')' + prefix + r'(' + \ 

490 r'|'.join(re.escape(x) for x in words) + r')' + suffix + r'(?!' + \ 

491 boundary_regex_fragment + r')'

493 def brackets_callback(token_class): 

494 def callback(lexer, match, context): 

495 groups = match.groupdict() 

496 opening_chars = groups['delimiter'] 

497 n_chars = len(opening_chars) 

498 adverbs = groups.get('adverbs') 

499 

500 closer = Perl6Lexer.PERL6_BRACKETS.get(opening_chars[0]) 

501 text = context.text 

502 

503 if closer is None: # it's not a mirrored character, which means we 

504 # just need to look for the next occurrence 

505 

506 end_pos = text.find(opening_chars, match.start('delimiter') + n_chars) 

507 else: # we need to look for the corresponding closing character, 

508 # keep nesting in mind 

509 closing_chars = closer * n_chars 

510 nesting_level = 1 

511 

512 search_pos = match.start('delimiter') 

513 

514 while nesting_level > 0: 

515 next_open_pos = text.find(opening_chars, search_pos + n_chars) 

516 next_close_pos = text.find(closing_chars, search_pos + n_chars) 

517 

518 if next_close_pos == -1: 

519 next_close_pos = len(text) 

520 nesting_level = 0 

521 elif next_open_pos != -1 and next_open_pos < next_close_pos: 

522 nesting_level += 1 

523 search_pos = next_open_pos 

524 else: # next_close_pos < next_open_pos 

525 nesting_level -= 1 

526 search_pos = next_close_pos 

527 

528 end_pos = next_close_pos 

529 

530 if end_pos < 0: # if we didn't find a closer, just highlight the 

531 # rest of the text in this class 

532 end_pos = len(text) 

533 

534 if adverbs is not None and re.search(r':to\b', adverbs): 

535 heredoc_terminator = text[match.start('delimiter') + n_chars:end_pos] 

536 end_heredoc = re.search(r'^\s*' + re.escape(heredoc_terminator) + 

537 r'\s*$', text[end_pos:], re.MULTILINE) 

538 

539 if end_heredoc: 

540 end_pos += end_heredoc.end() 

541 else: 

542 end_pos = len(text) 

543 

544 yield match.start(), token_class, text[match.start():end_pos + n_chars] 

545 context.pos = end_pos + n_chars 

546 

547 return callback

549 def opening_brace_callback(lexer, match, context): 

550 stack = context.stack 

551 

552 yield match.start(), Text, context.text[match.start():match.end()] 

553 context.pos = match.end() 

554 

555 # if we encounter an opening brace and we're one level 

556 # below a token state, it means we need to increment 

557 # the nesting level for braces so we know later when 

558 # we should return to the token rules. 

559 if len(stack) > 2 and stack[-2] == 'token': 

560 context.perl6_token_nesting_level += 1 

561 

562 def closing_brace_callback(lexer, match, context): 

563 stack = context.stack 

564 

565 yield match.start(), Text, context.text[match.start():match.end()] 

566 context.pos = match.end() 

567 

568 # if we encounter a free closing brace and we're one level 

569 # below a token state, it means we need to check the nesting 

570 # level to see if we need to return to the token state. 

571 if len(stack) > 2 and stack[-2] == 'token': 

572 context.perl6_token_nesting_level -= 1 

573 if context.perl6_token_nesting_level == 0: 

574 stack.pop()

576 def embedded_perl6_callback(lexer, match, context): 

577 context.perl6_token_nesting_level = 1 

578 yield match.start(), Text, context.text[match.start():match.end()] 

579 context.pos = match.end() 

580 context.stack.append('root')

666 def strip_pod(lines): 

667 in_pod = False 

668 stripped_lines = [] 

669 

670 for line in lines: 

671 if re.match(r'^=(?:end|cut)', line): 

672 in_pod = False 

673 elif re.match(r'^=\w+', line): 

674 in_pod = True 

675 elif not in_pod: 

676 stripped_lines.append(line) 

677 

678 return stripped_lines 

679 

680 # XXX handle block comments 

681 lines = text.splitlines() 

682 lines = strip_pod(lines) 

683 text = '\n'.join(lines) 

684 

685 if shebang_matches(text, r'perl6|rakudo|niecza|pugs'): 

686 return True 

687 

688 saw_perl_decl = False 

689 rating = False 

690 

691 # check for my/our/has declarations 

692 if re.search(r"(?:my|our|has)\s+(?:" + Perl6Lexer.PERL6_IDENTIFIER_RANGE + 

693 r"+\s+)?[$@%&(]", text): 

694 rating = 0.8 

695 saw_perl_decl = True 

696 

697 for line in lines: 

698 line = re.sub('#.*', '', line) 

699 if re.match(r'^\s*$', line): 

700 continue 

701 

702 # match v6; use v6; use v6.0; use v6.0.0; 

703 if re.match(r'^\s*(?:use\s+)?v6(?:\.\d(?:\.\d)?)?;', line): 

704 return True 

705 # match class, module, role, enum, grammar declarations 

706 class_decl = re.match(r'^\s*(?:(?P<scope>my|our)\s+)?(?:module|class|role|enum|grammar)', line) 

707 if class_decl: 

708 if saw_perl_decl or class_decl.group('scope') is not None: 

709 return True 

710 rating = 0.05 

711 continue 

712 break 

713 

714 return rating
